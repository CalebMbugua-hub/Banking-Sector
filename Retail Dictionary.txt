# 1) Create dictionary with 5 products and stock quantities
inventory = {
    "USB-C Cable": 25,
    "Laptop Charger": 8,
    "Wireless Mouse": 15,
    "HDMI Adapter": 5,
    "Webcam 1080p": 12
}
print("Initial inventory:", inventory)

# 2) Add a new product and update the quantity of an existing one
inventory["Bluetooth Speaker"] = 20    # add new
inventory["Laptop Charger"] = 12       # update existing from 8 to 12
print("\nAfter add/update:", inventory)

# 3) Function to return products with stock less than 10
def low_stock_products(inv, threshold=10):
    """Return a list of product names whose stock is less than threshold."""
    return [product for product, qty in inv.items() if qty < threshold]

print("\nLow stock products (threshold <10):", low_stock_products(inventory))

# 4) Delete a discontinued product and display updated dictionary
del inventory["HDMI Adapter"]       # delete discontinued product
print("\nInventory after deletion:", inventory)

# 5) Use .items() to loop through and print each product with its quantity
print("\nInventory listing:")
for product, qty in inventory.items():
    print(f" - {product}: {qty}")
